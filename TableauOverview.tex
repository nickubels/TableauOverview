\section{Tableaux for classical propositional logic}
Initial list for $A_1, ... A_n \models B$ is with $\neg B$. Closure condition: $A$ and $\neg A$ for some formula $A$ occur on a branch. \\
\Tree [.{$A \supset B$} [.{$\neg A$} ] [.{$B$} ] ]
\Tree [.{$\neg (A \supset B )$} [.{$A$} [.{$\neg B$} ] ] ]
\Tree [.{$A \vee B$} [.{$A$} ] [.{$B$} ] ]
\Tree [.{$\neg (A \vee B )$} [.{$\neg A$} [.{$\neg B$} ] ] ]
\Tree [.{$\neg \neg A$} [.{$A$} ] ]
\Tree [.{$A \wedge B $} [.{$A$} [.{$B$} ] ] ]
\Tree [.{$\neg(A \wedge B)$} [.{$\neg A$} ] [.{$\neg B$} ] ]
\Tree [.{$A \equiv B$} [.{$A$} [.{$B$} ] ] [.{$\neg A$} [.{$\neg B$} ] ] ]
\Tree [.{$\neg(A \equiv B)$} [.{$A$} [.{$\neg B$} ] ] [.{$\neg A$} [.{$B$} ] ] ]
\section{Semantic tableaux for FDE}
Initial list for $A_1, ... A_n \models B$ is with all $A_n, +$ and $B, -$. Closure condition: $A, +$ and $A, -$ for some formula $A$ occur on a branch. \\
\Tree [.{$A \wedge B, + $} [.{$A, +$} [.{$B, +$} ] ] ]
\Tree [.{$A \wedge B, -$} [.{$A, -$} ] [.{$B, -$} ] ]
\Tree [.{$A \vee B, +$} [.{$A, +$} ] [.{$B, +$} ] ]
\Tree [.{$A \vee B, -$} [.{$A, -$} [.{$B, -$} ] ] ]\\
\Tree [.{$\neg (A \wedge B), +$} [.{$\neg A \vee \neg B, +$} ] ]
\Tree [.{$\neg (A \wedge B), -$} [.{$\neg A \vee \neg B, -$} ] ]
\Tree [.{$\neg (A \vee B), +$} [.{$\neg A \wedge \neg B, +$} ] ]
\Tree [.{$\neg (A \vee B), -$} [.{$\neg A \wedge \neg B, -$} ] ]
\Tree [.{$\neg \neg A, +$} [.{$A, +$} ] ]
\Tree [.{$\neg \neg A, -$} [.{$A, -$} ] ]
\section{Semantic tableaux for $K_3, LP, L_3, RM_3$}
\subsection{$K_3$}
Same as for FDE with extra closure condition: A branch also closes if it contains $A, +$ and $\neg A, +$ for some formula $A$.
\subsection{$LP$}
Same as for FDE with extra closure condition: A branch also closes if it contains $A, -$ and $\neg A, -$ for some formula $A$.
\subsection{$L_3$}
Like $K_3$ but with the following rules for $\supset$:\\
\Tree [.{$A \supset B, +$} [.{$\neg A, +$} ] [.{$B, +$} ] [.{$A \vee \neg A, -$} [.{$B \vee \neg B, -$} ] ] ]
\Tree [.{$A \supset B, -$} [.{$A, +$} [.{$B, -$} ] ] [.{$\neg B, +$} [.{$\neg A, -$} ] ] ]
\Tree [.{$\neg(A \supset B), +$} [.{$A, +$} [.{$\neg B, +$} ] ] ]
\Tree [.{$\neg(A \supset B), -$} [.{$A, -$} ] [.{$\neg B,-$} ] ]
\subsection{$RM_3$}
Like $LP$ but with the following rules for $\supset$:\\
\Tree [.{$A \supset B, +$} [.{$A, -$} ] [.{$\neg B, -$} ] [.{$A \wedge \neg A, +$} [.{$B \wedge \neg B, +$} ] ] ]
\Tree [.{$A \supset B, -$} [.{$A, +$} [.{$B, -$} ] ] [.{$\neg B, +$} [.{$\neg A, -$} ] ] ]
\Tree [.{$\neg(A \supset B), +$} [.{$A, +$} [.{$\neg B, +$} ] ] ]
\Tree [.{$\neg(A \supset B), -$} [.{$A, -$} ] [.{$\neg B,-$} ] ]

\section{Tableaux for modal logic}
Initial list for $A_1, ... A_n \models B$ is with all $A_n, 0$ and $\neg B, 0$. Closure condition: $A, i$ and $\neg A, i$ for some formula $A$ occur on a branch with the same number $i$. \\
\Tree [.{$A \supset B, i$} [.{$\neg A, i$} ] [.{$B, i$} ] ]
\Tree [.{$\neg (A \supset B ), i$} [.{$A, i$} [.{$\neg B, i$} ] ] ]
\Tree [.{$A \vee B, i$} [.{$A, i$} ] [.{$B, i$} ] ]
\Tree [.{$\neg (A \vee B ), i$} [.{$\neg A, i$} [.{$\neg B, i$} ] ] ]
\Tree [.{$\neg \neg A, i$} [.{$A, i$} ] ]
\Tree [.{$A \wedge B, i$} [.{$A, i$} [.{$B, i$} ] ] ]
\Tree [.{$\neg(A \wedge B), i$} [.{$\neg A, i$} ] [.{$\neg B, i$} ] ]
\Tree [.{$A \equiv B, i$} [.{$A, i$} [.{$B, i$} ] ] [.{$\neg A, i$} [.{$\neg B, i$} ] ] ]\\
\Tree [.{$\neg(A \equiv B), i$} [.{$A, i$} [.{$\neg B, i$} ] ] [.{$\neg A, i$} [.{$B, i$} ] ] ]
\Tree [.{$\neg \Box A, i$} [.{$\Diamond \neg A,i$} ] ]
\Tree [.{$\neg \Diamond A,i$} [.{$\Box \neg A,i$} ] ]
\Tree [.{$\Box A, i$ \\ $irj$} [.{$A,j$} ] ]
\Tree [.{$\Diamond A, i$} [.{$irj$ \\ $A,j$} ] ]
\Tree [.{$\neg \Box A,i$} [.{$irj$ \\ $\neg A, j$} ] ]
\Tree [.{$\neg \Diamond A, i$ \\ $irj$} [.{$\neg A, j$} ] ]

\end{minipage}%
\begin{minipage}[t]{.33\textwidth}
Dense ($\delta$):\\
\Tree [.{$irj$} [.{$irk$ \\ $krj$} ] ]\\
\end{minipage}\\
\begin{minipage}[t]{.33\textwidth}
Extra rules for $=$:\\
\Tree [.{$\alpha(i)$ \\ $i = j$} [.{$\alpha(j)$} ] ]
\Tree [.{$\alpha(i)$ \\ $j = i$} [.{$\alpha(j)$} ] ]\\
\end{minipage}%
\begin{minipage}[t]{.33\textwidth}
Forward convergent ($\varphi$): \\
\Tree [.{$irj$ \\ $irk$} [.{$jrk$} ] [.{$j = k$} ] [.{$krj$} ] ]\\
\end{minipage}%
\begin{minipage}[t]{.33\textwidth}
Backward convergent ($\beta$): \\
\Tree [.{$jri$ \\ $kri$} [.{$jrk$} ] [.{$j = k$} ] [.{$krj$} ] ]
\end{minipage}

\section{First-order modal logic}
\subsection{$CK$}
\Tree [.{$\neg \exists xA,i$} [.{$\forall x \neg A,i$} ] ]
\Tree [.{$\neg \forall xA,i$} [.{$\exists x \neg A,i$} ] ]
\Tree [.{$\forall xA,i$} [.{$A_x(a),i$ \\ where $a$ is old (if possible);\\ otherwise introduce an $a$} ] ]
\Tree [.{$\exists xA,i$} [.{$A_x(c),i$ \\ where $c$ is new} ] ]
\subsection{$VK$}
\Tree [.{$\forall xA,i$} [.{$\neg \mathfrak{E} A,i$} ] [.{$A_x(a),i$} ] ]
\Tree [.{$\exists xA,i$} [.{$\mathfrak{E}c,i$} [.{$A_x(c),i$} ] ] ]\\
$a$ is old, if possible; otherwise introduce a new $a$,
$c$ is a new variable
